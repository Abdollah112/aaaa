      { id: 4, name: 'OnePlus 11', price: 699, type: 'Smartphone', imageUrl: 'https://s.abcnews.com/images/International/gty_temple_lb_150824_16x9_992.jpg?w=992' },

    ];

    window.selectedProduct = null;

    // Function to render the product list
    function renderProductList() {
      const productListElement = document.getElementById('productList');
      productListElement.innerHTML = ''; // Clear existing items

      window.products.forEach(product => {
        const li = document.createElement('li');
        li.onclick = function () { selectProduct(product); };

        const img = document.createElement('img');
        img.src = product.imageUrl;
        img.alt = product.name;

        const p = document.createElement('p');
        p.textContent = product.name;

        li.appendChild(img);
        li.appendChild(p);
        productListElement.appendChild(li);
      });
    }

    // Function to select a product
    function selectProduct(product) {
      window.selectedProduct = product;
      renderProductDetails();
    }

    // Function to render the selected product details
    function renderProductDetails() {
      if (window.selectedProduct) {
        const detailsElement = document.getElementById('productDetails');
        const productImage = document.getElementById('productImage');
        const productName = document.getElementById('productName');
        const productPrice = document.getElementById('productPrice');
        const productType = document.getElementById('productType');

        productImage.src = window.selectedProduct.imageUrl;
        productImage.alt = window.selectedProduct.name;
        productName.textContent = window.selectedProduct.name;
        productPrice.textContent = `$${window.selectedProduct.price}`;
        productType.textContent = window.selectedProduct.type;

        detailsElement.style.display = 'block'; // Show product details
      }
    }

    // Initial render of the product list
    renderProductList();
</script>
